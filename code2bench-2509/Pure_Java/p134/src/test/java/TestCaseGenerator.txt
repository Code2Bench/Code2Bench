package p134;
import com.google.gson.Gson;
import com.google.gson.GsonBuilder;
import java.io.*;
import java.util.*;
import java.util.concurrent.atomic.AtomicInteger;
import net.jqwik.api.*;
import net.jqwik.api.lifecycle.AfterContainer;
import p134.GroundTruth;

public class TestCaseGenerator {

    static class TestCase {
        Map<String, Object> Inputs;
        Object Expected;
        TestCase(Map<String, Object> inputs, Object expected) {
            this.Inputs = inputs;
            this.Expected = expected;
        }
    }

    static List<TestCase> generatedCases = new ArrayList<>();
    static AtomicInteger caseCount = new AtomicInteger(0);
    static final int MAX_CASES = 500;

    @Property(tries = 10000)
    void generateTestCases(@ForAll("codePoint") int codePoint) {
        if (caseCount.get() >= MAX_CASES) {
            return;
        }

        boolean result;
        try {
            result = GroundTruth.isValidIdentifierPart(codePoint);
        } catch (Exception e) {
            return;
        }

        Map<String, Object> inputs = new LinkedHashMap<>();
        inputs.put("codePoint", codePoint);
        generatedCases.add(new TestCase(inputs, result));
        caseCount.incrementAndGet();
    }

    @Example
    void example1() {
        generateTestCases(65); // 'A'
    }

    @Example
    void example2() {
        generateTestCases(64); // '@'
    }

    @Example
    void example3() {
        generateTestCases(46); // '.'
    }

    @Example
    void example4() {
        generateTestCases(45); // '-'
    }

    @Example
    void example5() {
        generateTestCases(43); // '+'
    }

    @Example
    void example6() {
        generateTestCases(61); // '='
    }

    @Example
    void example7() {
        generateTestCases(47); // '/'
    }

    @Example
    void example8() {
        generateTestCases(35); // '#'
    }

    @Example
    void example9() {
        generateTestCases(0);  // Null character
    }

    @Example
    void example10() {
        generateTestCases(55296); // Surrogate pair start
    }

    @Example
    void example11() {
        generateTestCases(57343); // Surrogate pair end
    }

    @Example
    void example12() {
        generateTestCases(65536); // Supplementary character
    }

    @Example
    void example13() {
        generateTestCases(97); // 'a'
    }

    @Example
    void example14() {
        generateTestCases(48); // '0'
    }

    @Example
    void example15() {
        generateTestCases(95); // '_'
    }

    @Example
    void example16() {
        generateTestCases(128512); // Emoji
    }

    @Provide
    Arbitrary<Integer> codePoint() {
        return Arbitraries.integers().between(0, 1114111); // Full Unicode range
    }

    @AfterContainer
    static void saveTestCases() throws IOException {
        if (generatedCases.isEmpty()) {
            return;
        }
        String dirPath = "src/test/java/test_cases";
        File dir = new File(dirPath);
        if (!dir.exists()) {
            dir.mkdirs();
        }
        String filePath = dirPath + "/test_cases.json";
        Gson gson = new GsonBuilder().setPrettyPrinting().create();
        try (Writer writer = new FileWriter(filePath)) {
            gson.toJson(generatedCases, writer);
        }
        System.out.println("âœ… Saved " + generatedCases.size() + " test cases to " + filePath);
    }
}
