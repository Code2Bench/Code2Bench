package p93;
import com.google.gson.Gson;
import com.google.gson.GsonBuilder;
import java.io.*;
import java.util.*;
import java.util.concurrent.atomic.AtomicInteger;
import net.jqwik.api.*;
import net.jqwik.api.lifecycle.AfterContainer;
import p93.GroundTruth;

public class TestCaseGenerator {

    static class TestCase {
        Map<String, Object> Inputs;
        Object Expected;
        TestCase(Map<String, Object> inputs, Object expected) {
            this.Inputs = inputs;
            this.Expected = expected;
        }
    }

    static List<TestCase> generatedCases = new ArrayList<>();
    static AtomicInteger caseCount = new AtomicInteger(0);
    static final int MAX_CASES = 500;

    @Property(tries = 10000)
    void generateTestCases(@ForAll("string") String currentOutput, @ForAll("string") String lastOutput) {
        if (caseCount.get() >= MAX_CASES) {
            return;
        }

        String result;
        try {
            result = GroundTruth.calculateIncrementalContent(currentOutput, lastOutput);
        } catch (Exception e) {
            return;
        }

        boolean isPrefixBranch = lastOutput != null && !lastOutput.isEmpty() && currentOutput.startsWith(lastOutput);
        boolean isElseBranch = !isPrefixBranch;
        if (!isPrefixBranch && !isElseBranch) {
            return;
        }

        Map<String, Object> inputs = new LinkedHashMap<>();
        inputs.put("currentOutput", currentOutput);
        inputs.put("lastOutput", lastOutput);
        generatedCases.add(new TestCase(inputs, result));
        caseCount.incrementAndGet();
    }

    @Example
    void example1() {
        generateTestCases("hello", "hell");
    }

    @Example
    void example2() {
        generateTestCases("world", "");
    }

    @Example
    void example3() {
        generateTestCases("test", "test");
    }

    @Example
    void example4() {
        generateTestCases("", "");
    }

    @Example
    void example5() {
        generateTestCases("abc", "xyz");
    }

    @Example
    void example6() {
        generateTestCases("prefix", "pre");
    }

    @Example
    void example7() {
        generateTestCases("suffix", "fix");
    }

    @Example
    void example8() {
        generateTestCases("", "nonEmpty");
    }

    @Example
    void example9() {
        generateTestCases("nonEmpty", null);
    }

    @Example
    void example10() {
        generateTestCases("nullPrefix", null);
    }

    @Provide
    Arbitrary<String> string() {
        return Arbitraries.strings().withChars("abcdefghijklmnopqrstuvwxyz").ofMinLength(0).ofMaxLength(10);
    }

    @AfterContainer
    static void saveTestCases() throws IOException {
        if (generatedCases.isEmpty()) {
            return;
        }
        String dirPath = "src/test/java/test_cases";
        File dir = new File(dirPath);
        if (!dir.exists()) {
            dir.mkdirs();
        }
        String filePath = dirPath + "/test_cases.json";
        Gson gson = new GsonBuilder().setPrettyPrinting().create();
        try (Writer writer = new FileWriter(filePath)) {
            gson.toJson(generatedCases, writer);
        }
        System.out.println("âœ… Saved " + generatedCases.size() + " test cases to " + filePath);
    }
}
